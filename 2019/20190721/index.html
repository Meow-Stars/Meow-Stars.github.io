代码1

创建swift文件扩展UserDefaults

import Foundation
import UIKit

extension UserDefaults {
    // MARK: 应用第一次启用
    static func isFirstLaunch() -> Bool {
        let hasBeenLaunched = "hasBeenLaunched"
        let isFirstLaunch = !UserDefaults.standard.bool(forKey: hasBeenLaunched)
        if isFirstLaunch {
            UserDefaults.standard.set(true, forKey: hasBeenLaunched)
            UserDefaults.standard.synchronize()
        }
        return isFirstLaunch
    }
}

代码2

打开文件AppDelegate.swift，修改函数application()

func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
    // 判断APP第一次启动
    if UserDefaults.isFirstLaunch() {
        print("这是第一次启动")
        let initialLaunch = UIStoryboard(name: "InitialLaunch", bundle: nil).instantiateViewController(withIdentifier: "InitialLaunch") as UIViewController
        self.window!.rootViewController = initialLaunch
    }
    return true
}
