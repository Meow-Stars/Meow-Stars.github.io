怎么切换不同的UIPickerView？

为每一个UIPickerView设立tag let periodDaysPicker = UIPickerView() periodDaysPicker.delegate = self periodDaysPicker.dataSource = self periodDaysPicker.tag = 3

// 每个选项显示什么 func pickerView(_ pickerView: UIPickerView, titleForRow row: Int, forComponent component: Int) -> String? { switch pickerView.tag { case 2: return lastDaysTextOptions[row] case 3: return periodDaysTextOptions[row] default: return “没有选择器” } }

第一步：在viewDidLoad()里写： // MARK: 点击空白隐藏键盘 let tap = UITapGestureRecognizer(target: self, action: #selector(InitialLaunch.hideKeyboard(tapG:))) tap.cancelsTouchesInView = false self.view.addGestureRecognizer(tap)
