【线性表】带头结点的单链表——基础操作

Written by 云深隐星澜 with ♥ on September 7, 2020 in 数据结构

定义结点类型

typedef int Elemtype;
// 因为该结构体中包含了Node类型变量（指针域），这是结构体的自引用，只能通过自引用指针*LinkList实现
typedef struct Node{
    Elemtype data;		// 数据域
    struct Node *next;	// 指针域
}Node,*LinkList;

创建单链表

头插法

头插法是将结点a~n~、a~n-1~、a~n-2~、……、a~1~逐个插入到头结点后面，因此输入时要逆序输入

LinkList CreateLinkListHead(int n){
    // 先建立一个带头结点的空单链表L
    Node *L;
    L = (Node *)malloc(sizeof(Node));
    L -> next = NULL;
    // 逆序输入n个元素的值
    printf("输入n个元素的值：");
    for(int i = n; i > 0; i--){
        Node *p;
        p = (Node *)malloc(sizeof(Node));	      // 申请新的结点空间
        scanf("%d",&p -> data);					// 输入元素值
        p -> next = L -> next;				 // 将新结点插入到表头
        L -> next = p;
    }
    return L;
}

尾插法

尾插法是将结点插入到最后一个结点后面，即新插入的结点变为链表最后一个结点

LinkList CreateLinkListTail(int n){
    // 先建立一个带头结点的空单链表L，pL是指向头结点的指针
    Node *L;
    L = (Node *)malloc(sizeof(Node));
    L -> next = NULL;
    Node *pL = L;
    // 顺序输入n个元素的值
    printf("输入n个元素的值：");
    for(int i = 0; i < n; i++){
        Node *p;
        p = (Node *)malloc(sizeof(Node));	      // 申请新的结点空间
        scanf("%d",&p -> data);					// 输入元素值
        p -> next = NULL;
        pL -> next = p;                            // 将新结点插入到表头
        pL = pL -> next;
    }
    return L;
}

插入

在链表的第i个位置插入元素e，使元素e称为新链表的第i个元素

int ListInsert(LinkList L, int i, Elemtype e){
    Node *p;
    p = L;			// p指向第i个位置结点的前驱结点，初始结果 = 头结点
    int index ;	// 计数器
    for(index = 1; index < i; index++){		// 寻找第i-1个结点，即第i个结点的前驱结点
        p = p -> next;
    }
    Node *s;
    s = (Node *)malloc(sizeof(Node));
    s -> data = e;							    // 插入值为e的结点x
    s -> next = p -> next;
    p -> next = s;
    return 1;
}

删除

删除链表中值为e的结点，这段代码如果有重复元素，只能删除第一个出现的

// 单链表的删除，删除链表中值为e的元素
int ListDelete(LinkList L, Elemtype e){
    Node *p,*x;				// p指向待删除结点的前驱结点，x指向待删除结点
    p = L ;
    x = L -> next;
    while(x -> data != e){
        p = x;
        x = x -> next;
    }
    p -> next = x -> next;	// 删除并释放结点x
    free(x);
    return 1;
}

查找

在单链表中查找元素e，如果存在元素e，返回e第一次出现的位置，否则返回-1

// 在链表中查找结点e
int ListSearch(LinkList L, Elemtype e){
    Node *p = L;                // p最先指向头结点
    int i = 1;                    // i是结点e的位置
    while(p -> next){
        p = p -> next;          // 为什么要先执行p = p -> next？因为要让p指向第一个结点
        if(p -> data == e){
            return i;
        }
        i++;
    }
    return -1;                  // 不存在结点e则返回-1
}

获取单链表长度

这个长度是单链表有意义的长度，不包含头结点

int ListLength(LinkList L){
    Node *p = L;
    int index = 0;
    while(p -> next){
        p = p -> next;
        index++;
    }
    return index;
}

打印整个单链表

void PrintList(LinkList list){
    Node *p = list;
    if( p -> next == NULL){             // p指向list的头结点，所以判断空表是p -> next == NULL
        printf("这素一个空链表");
    }
    while(p -> next){
        p = p -> next;
        printf("%d  ",p -> data);
    }
    printf("\n");
}

main( )函数

int main(){
	LinkList list;
	// 创建单链表A
	printf("输入链表长度n：");
	int n;
	scanf("%d",&n);
	list = CreateLinkListTail(n);
	PrintList(list);
	printf("单链表长度%d",ListLength(list));

	// 插入结点
	printf("输入插入位置i：");
	int i;
	scanf("%d",&i);
	printf("输入插入元素e的值：");
	Elemtype e1;
	scanf("%d",&e1);
	ListInsert(list,i,e1);
	PrintList(list);

	// 删除结点
	printf("输入删除元素e的值：");
	Elemtype e2;
	scanf("%d",&e2);
	ListDelete(list,e2);
    PrintList(list);

    // 查找结点
    printf("输入查找元素e的值：");
    Elemtype e3;
    scanf("%d",&e3);
    int index = ListSearch(list,e3);
    printf("%d");
}
