【iOS】UICollectionView中使用Button

Written by 云深隐星澜 with ♥ on March 5, 2020 in iOS•Swift

我的每一个item上都有一个Button，我想点击按钮就能知道这是第几个item。

单独item的设置card.swift

import UIKit
protocol GetProtocol {
    func didGetProtocol(sender: Card)
}
class Card: UICollectionViewCell {
    var titleLabel: UILabel!
    var answerButton: UIButton!
    // 遵循协议
    var delegate: GetProtocol?
    
    override init(frame: CGRect) {
        super.init(frame: frame)
        // 获取item大小
        let itemWidth: CGFloat = self.frame.size.width
        let itemHeight: CGFloat = self.frame.size.height
        // 添加按钮
        answerButton = UIButton(type: .system)
        answerButton.frame = CGRect(frame: CGRect(x: itemWidth*0.1, y: 5, width: itemWidth*0.8, height: 30))

        answerButton.addTarget(self, action: #selector(selected(button: )), for: .touchDown)
        self.addSubview(answerButton)
    }

    }
}

UICollectionView的设置collectionview.swift

import UIKit
class Dimension1Answer: UIViewController,UICollectionViewDelegate,UICollectionViewDataSource,GetProtocol {
    var collectionView: UICollectionView!
    var flowLayout: UICollectionViewFlowLayout!
    let itemIdentify = "Item"
    
    override func viewDidLoad() {
        super.viewDidLoad()
        addCollectionView()
        registerItem()
    }
    // MRAK: 实现UICollectionView
    func addCollectionView(){
        // item的布局与尺寸
        flowLayout = UICollectionViewFlowLayout()      
        flowLayout.itemSize = CGSize(width: fullScreenSize.width, height: fullScreenSize.height*0.2)                 
        flowLayout.scrollDirection = .vertical      
        flowLayout.minimumInteritemSpacing = 40     // 行间距
        // 创建UICollectionView
        let collectionViewFrame: CGRect = CGRect(x: 0, y: fullScreenSize.height*0.3, width: fullScreenSize.width, height: fullScreenSize.height*0.65)
        collectionView = UICollectionView(frame: collectionViewFrame, collectionViewLayout: flowLayout)
        collectionView.backgroundColor = UIColor.orange
        collectionView.dataSource = self
        collectionView.delegate = self
        self.view.addSubview(collectionView)
    }
    // 注册item，使之重用
    func registerItem(){
        collectionView.register(Card.self, forCellWithReuseIdentifier: itemIdentify)
    }
    // item数量
    func collectionView(_ collectionView: UICollectionView, numberOfItemsInSection section: Int) -> Int {
        return 5
    }
    // item内容
    func collectionView(_ collectionView: UICollectionView, cellForItemAt indexPath: IndexPath) -> UICollectionViewCell {
        let item: Card? = collectionView.dequeueReusableCell(withReuseIdentifier: itemIdentify, for: indexPath) as? Card
        // MARK: delegate成功的关键！！！
        item?.delegate = self
        return item!
    } 
    // MARK: 实现GetProtocol
    func didGetScore(sender: Card){
        print("我是协议")
        // 获取点击的是第几个item
        let point: CGPoint = sender.center
        let index: IndexPath = collectionView.indexPathForItem(at: point)!
        print("点击了第\(index.row+1)个item")	// 序号是从0开始的
    }
}
