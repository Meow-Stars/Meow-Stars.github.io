【iOS】创建一个ToDoList APP(二)

上一篇是静态的，cell的数目、显示的内容都是固定的，现在将其扩展成动态，可以实现添加、删除

新建按钮，跳转输入界面

为了美观，就在主界面的导航栏右侧新增了一个按钮

  • 新建UIViewController，取名AddScreen

  • AppDelegate.swift，注意要设置StoryboardID

func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
    // 添加对导航栏的支持
    let initialLaunch = UIStoryboard(name: "Main", bundle: nil).instantiateViewController(withIdentifier: "ViewController") as UIViewController
    let navigationVC = UINavigationController(rootViewController: initialLaunch)
    self.window!.rootViewController = navigationVC
    return true
}
  • ViewController.swift，创建导航栏
func setNavigationBar(){
    self.title = "ToDoList"
    self.navigationItem.rightBarButtonItem = UIBarButtonItem(barButtonSystemItem: UIBarButtonItem.SystemItem.add, target: self, action: #selector(AddItem))
}
// Add按钮
@objc func AddItem(){
    let AddScreenVC = AddScreen()
    AddScreenVC.delegate = self
    self.present(AddScreenVC, animated: true, completion: nil)
}

AddScreen.swift

  • 添加协议
protocol AddProtocal {
    func didCompleted(addObject: AddScreen)
}

  • 添加控件，一个UILabel，一个UITextField，一个UIButton

  • 如果输入框为空，点击按钮输入框会抖动

完整AddScreen.swift

import UIKit

protocol AddProtocal {
    func didCompleted(addObject: AddScreen)
}

class AddScreen: UIViewController,UITextFieldDelegate {
    var toDoLabel: UILabel!
    var toDoField: UITextField!
    var completeButton: UIButton!
    var delegate: AddProtocal?
    
    override func viewDidLoad() {
        super.viewDidLoad()
        self.view.backgroundColor = UIColor.white
        addUI()
    }
    required init?(coder aDecoder: NSCoder) {
        super.init(coder: aDecoder)
    }
    override init(nibName nibNameOrNil: String?, bundle nibBundleOrNil: Bundle?) {
        super.init(nibName: nibNameOrNil, bundle: nibBundleOrNil)
    }
    override func viewWillAppear(_ animated: Bool) {
        setUp()
    }
    func addUI(){
    	// UILabel
        toDoLabel = UILabel(frame: CGRect(x: fullScreenSize.width*0.15, y: fullScreenSize.height*0.05, width: fullScreenSize.width*0.7, height: fullScreenSize.height*0.1))
        toDoLabel.text = "ToDo"
        toDoLabel.font = UIFont.systemFont(ofSize: 20)
        self.view.addSubview(toDoLabel)
        // UITextField
        toDoField = UITextField(frame: CGRect(x: fullScreenSize.width*0.15, y: fullScreenSize.height*0.1, width: fullScreenSize.width*0.7, height: fullScreenSize.height*0.3))
        toDoField.placeholder = "请输入待办事项"            // 提示语
        toDoField.delegate = self
        toDoField.font = UIFont.systemFont(ofSize: 20)
        self.view.addSubview(toDoField)
        // UIButton
        completeButton = UIButton(type: .system)
        completeButton.frame = CGRect(x: fullScreenSize.width*0.2, y: fullScreenSize.height*0.7, width: fullScreenSize.width*0.6, height: 50)
        completeButton.layer.cornerRadius = 5.0
        completeButton.backgroundColor = UIColor.blue
        completeButton.addTarget(self, action: #selector(CompleteType(sender:)), for: .touchDown)
        self.view.addSubview(completeButton)
    }
    @objc func CompleteType(sender: AnyObject){
        // 完成输入
        if toDoField.text!.count > 0{
            delegate?.didCompleted(addObject: self)
            self.dismiss(animated: true, completion: nil)
        }
        // 如果输入为空则输入框抖动
        else{   
            shakeAnimation(sender: toDoField)
        }
    }
    func setUp(){
        if delegate == nil{
            toDoField.textColor = UIColor.lightGray
            toDoField.isUserInteractionEnabled = false
            completeButton.setTitle("完成", for: .normal)
        }
        else{
            toDoField.textColor = UIColor.black
            toDoField.isUserInteractionEnabled = true
            completeButton.setTitle("OK", for: .normal)
        }
    }
    //  UITextField点击右下角回收键盘
    func textFieldShouldReturn(_ textField: UITextField) -> Bool {
        textField.resignFirstResponder()
        return true
    }
    // 没有输入则输入框抖动动画
    func shakeAnimation(sender: AnyObject){
        let animation = CAKeyframeAnimation()
        animation.keyPath = "position.x"
        animation.values = [0,10,-10,10,0]
        animation.keyTimes = [0,1/6.0,3/6.0,5/6.0,1] as [NSNumber]
        animation.duration = 0.4
        animation.isAdditive = true
        sender.layer?.add(animation, forKey: "shake")
    }
}

完整ViewController.swift

import UIKit

class ViewController: UIViewController,UITableViewDataSource,UITableViewDelegate,AddProtocal {
    var tableView: UITableView!
    let cellIdentify = "Cell"
    var toDoData = [AddScreen]()
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setNavigationBar()
        setTableView()
        registerCell()
    }
  
    // 创建导航栏
    func setNavigationBar(){
        self.title = "ToDoList"
        self.navigationItem.rightBarButtonItem = UIBarButtonItem(barButtonSystemItem: UIBarButtonItem.SystemItem.add, target: self, action: #selector(AddItem))
    }
    // Add按钮
    @objc func AddItem(){
        let AddScreenVC = AddScreen()
        AddScreenVC.delegate = self
        self.present(AddScreenVC, animated: true, completion: nil)
    }
    override func viewWillAppear(_ animated: Bool) {
    	// 导航栏不透明
        super.viewWillAppear(animated)
        self.navigationController?.navigationBar.isTranslucent = false  
    }
    // 创建UITableView
    func setTableView(){
        self.tableView = UITableView(frame: self.view.frame, style: .plain)
        self.tableView.delegate = self
        self.tableView.dataSource = self
        self.view.addSubview(tableView)
    }
    // 注册Cell
    func registerCell(){
        tableView.register(Cell.self, forCellReuseIdentifier: cellIdentify)
    }
    // 设置分区section数目
    func numberOfSections(in tableView: UITableView) -> Int {
        return 1
    }
    // 设置cell高度
    func tableView(_ tableView: UITableView, heightForRowAt indexPath: IndexPath) -> CGFloat {
        return 60
    }
    // 设置cell数目
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return toDoData.count
    }
    // 设置cell内容
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        var cell = tableView.dequeueReusableCell(withIdentifier: cellIdentify) as? Cell
        if cell == nil{
            let nibs:Array = Bundle.main.loadNibNamed("Cell", owner: self, options: nil)!
            cell = (nibs.last as? Cell)!
        }
        let addObject = toDoData[indexPath.row] as AddScreen
        cell?.todoTitle.text = addObject.toDoField.text
        cell?.accessoryType = UITableViewCell.AccessoryType.disclosureIndicator
        return cell!
    }
    func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
        let AddScreenVC = toDoData[indexPath.row] as AddScreen
        AddScreenVC.delegate = nil
        self.present(AddScreenVC, animated: true, completion: nil)
    }
    // 设置cell可以滑动删除
    func tableView(_ tableView: UITableView, commit editingStyle: UITableViewCell.EditingStyle, forRowAt indexPath: IndexPath) {
        if editingStyle == UITableViewCell.EditingStyle.delete{
            print("删除第\(indexPath.row+1)个选项")
            toDoData.remove(at: indexPath.row)
            tableView.reloadData()
        }
    }
    // 完成protocal中的函数
    func didCompleted(addObject: AddScreen) {
        toDoData.append(addObject)
        tableView.reloadData()
    }
    override func didReceiveMemoryWarning() {
        super.didReceiveMemoryWarning()
    }
}
