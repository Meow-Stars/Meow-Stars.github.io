计算2^N(N很大)

Written by 云深隐星澜 with ♥ on February 23, 2020 in

完整代码

#include <iostream>
using namespace std;

void calculatePower(int N){
	// 定义一个足够大的数组，并将除array[0]外全置0
	int array[N];		
	for(int i = 0; i < N; i++){
		array[i] = 0;
	}
	array[0] = 1;
	// 计算N^N
	for(int p = 0; p < N; p++){
		for(int j = 0; j < N; j++){
			array[j] *= 2;
		}
		for(int m = 0; m < N; m++){
			if(array[m] > 9){
				array[m+1]++;
				array[m] = array[m]%10;
			}
		}
	}
	// 逆序找出数组中第一个不为0的元素
	int q = 0;
	for(q = 0; q < N; q++){
		if(array[N-q-1] != 0){
			break;
		}
	}
	// 逆序输出
	for(; q < N; q++){
		cout << array[N-q-1];
	}
}

int main()
{
	int N;
	cout << "请输入幂：" << endl;
	cin >> N;
	cout << "2^" << N << "结果为：" << endl;
	calculatePower(N);
	cout << endl;
	return 0;
}
