【iOS】创建一个ToDoList APP(一)

Written by 云深隐星澜 with ♥ on January 31, 2020 in iOS•Swift

自定义cell

  • 创建一个Cocoa Touch Class文件，继承自UITableViewCell

  • 创建Cell类

import UIKit
class Cell: UITableViewCell{
    var todoTitle: UILabel!

    override init(style: UITableViewCell.CellStyle, reuseIdentifier: String?) {
        super.init(style: style, reuseIdentifier: reuseIdentifier)
        self.setUpCellUI()
    }
    func setUpCellUI(){
        todoTitle = UILabel(frame: CGRect(x: 30, y: 10, width: 200, height: 30))
        todoTitle.font = UIFont.systemFont(ofSize: 20)
        self.addSubview(todoTitle!)
        
    }
    required init?(coder aDecoder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
}

创建UITableView

  • 进入ViewController.swift，创建UITableView
import UIKit
class ViewController: UIViewController,UITableViewDataSource,UITableViewDelegate {
    var tableView: UITableView!
    let cellIdentify = "Cell"

    override func viewDidLoad() {
        super.viewDidLoad()

        self.tableView = UITableView(frame: self.view.frame, style: .plain)
        // 将代理、数据来源设置为自己
        self.tableView.delegate = self
        self.tableView.dataSource = self
        self.view.addSubview(tableView)
		// 注册Cell
        tableView.register(Cell.self, forCellReuseIdentifier: cellIdentify)
    }
}
  • 设置分区section的数目，如果不设则默认为1
func numberOfSections(in tableView: UITableView) -> Int {
    return 1
}
  • 设置cell的数目和内容，这两个函数是必须的
func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
    return 5
}
   
func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
    var cell = tableView.dequeueReusableCell(withIdentifier: cellIdentify) as? Cell
    if cell == nil{
        let nibs:Array = Bundle.main.loadNibNamed("Cell", owner: self, options: nil)!
        cell = (nibs.last as? Cell)!
    }
    cell!.todoTitle.text = "ToDoTitle"
    cell!.accessoryType = UITableViewCell.AccessoryType.disclosureIndicator
    return cell!
}
  • 设置cell的高度，这个是根据Cell类中控件大小定义的
func tableView(_ tableView: UITableView, heightForRowAt indexPath: IndexPath) -> CGFloat {
    return 60
}
  • 设置左滑显示“删除”，但是还不能真正实现删除哦
func tableView(_ tableView: UITableView, commit editingStyle: UITableViewCell.EditingStyle, forRowAt indexPath: IndexPath) {
    if editingStyle == UITableViewCell.EditingStyle.delete{
        print("删除第\(indexPath.row+1)个选项")
    }
}
